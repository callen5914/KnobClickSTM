cmake_minimum_required(VERSION 3.21)
if (${TOOLCHAIN_LANGUAGE} MATCHES "MikroC")
    project(lib_knob LANGUAGES MikroC)
else()
    project(lib_knob LANGUAGES C ASM)
endif()

if (NOT PROJECT_TYPE)
    set(PROJECT_TYPE "mikrosdk" CACHE STRING "" FORCE)
endif()

add_library(lib_knob STATIC
        src/knob.c
        include/knob.h
)
add_library(Click.Knob  ALIAS lib_knob)



target_include_directories(lib_knob PUBLIC
        include
)

find_package(MikroC.Core REQUIRED)
target_link_libraries(lib_knob PUBLIC MikroC.Core)
find_package(MikroSDK.Driver REQUIRED)
target_link_libraries(lib_knob PUBLIC MikroSDK.Driver)
